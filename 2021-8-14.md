# 10의 약수 출력

for i in range(1,11,1):

```python
# 입력한 숫자의 약수 출력

a = int(input('숫자: '))

for i in range(1,a+1,1):
    if a%i == 0:
        print(i,end=' ')

```


```python
# 두수의 공약수 출력

a = int(input('숫자1: '))
b = int(input('숫자2: '))

if a>b:
    a,b = b,a

for i in range(1,a+1,1):
    if a%i == 0 and b%i == 0:
        print(i,end=' ')

```


```python
# 두수의 공약수 출력

a = int(input('숫자1: '))
b = int(input('숫자2: '))
gcd = []

if a>b:
    a,b = b,a

for i in range(1,a+1,1):
    if a%i == 0 and b%i == 0:
        gcd.append(i)

print(max(gcd))
```
